// Criação da tabela
CREATE TABLE registro
(
  id int NOT NULL AUTO_INCREMENT,
  descricao VARCHAR(255) NOT NULL,
  valor DOUBLE NOT NULL,
  tipo VARCHAR(10) NOT NULL,
  PRIMARY KEY (id)
);
